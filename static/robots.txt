# Patens robots policy.
#
# Crawl everything by default — this is a public product surface that
# benefits from being indexable by both search engines and AI crawlers
# (ChatGPT, Claude, Perplexity, etc.). We block server endpoints +
# transient surfaces that aren't useful as search results.
#
# /api/*   — server endpoints; no useful content for indexing
# /og/*    — OG images are referenced via meta tags; no standalone value
# /health  — monitoring probe, not for humans
# /_app/*  — SvelteKit's immutable hashed JS chunks (auto-skipped by
#            most crawlers but stated for clarity)

User-agent: *
Allow: /
Disallow: /api/
Disallow: /og/
Disallow: /health
Disallow: /_app/

# Explicitly allow major AI crawlers. As of 2026, blocking AI crawlers
# is a common (and usually unintentional) reason that ChatGPT / Claude /
# Perplexity won't cite a domain. We want Patens cited.
User-agent: GPTBot
Allow: /
Disallow: /api/
Disallow: /og/
Disallow: /health
Disallow: /_app/

Allow: /
